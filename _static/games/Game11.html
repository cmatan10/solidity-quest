          pragma solidity ^0.8.10;
          
          contract EncodeData {
              bytes public _encodeStringAndUint = hex"";
          
              function encode(bytes memory encodedData) external {
                  require(
                      keccak256(encodedData) == keccak256(abi.encode("WEB", 3)),
                      "The Answer is incorrect"
                  );
                  _encodeStringAndUint = encodedData;
              }  
          }
Game Description

Master data encryption in Solidity. Use encoding tools and craft the perfect solution. You need to be familiar with the abi.encode function for encoding data in Solidity, understand how keccak256 works, and use these tools to encrypt data.
