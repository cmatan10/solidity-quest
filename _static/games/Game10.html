          pragma solidity ^0.8.10;
          
          contract InterfaceId {
              bool public answer;
          
              function CalcMe(bytes4 id) external {
                  require(id == bytes4(keccak256("CalcMe(bytes4)")), "Calc Me Again!");
                  answer = true;
              }
          }
Game Description

Decipher function signatures in Solidity. Compute and match using cryptographic functions. You need to understand function signatures in Solidity, how to compute them using keccak256, and convert them into bytes4 type.
